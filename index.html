    function parseUrls(text) {

      const value = String(text || '').trim();

      if (!value) return [];

      return Array.from(new Set(

        value

          .split(/[\n,]+/)

          .map((item) => item.trim())

          .filter(Boolean)

          .filter((item) => /^https?:\/\//i.test(item))

      ));

    }

    async function fetchUrl(url) {

      try {

        const res = await fetch(url, { mode: 'cors' });

        const text = await res.text();

        const doc = new DOMParser().parseFromString(text, 'text/html');

        const title = doc.querySelector('title')?.textContent?.trim() || '';

        const paragraphs = Array.from(doc.querySelectorAll('p,li,h1,h2,h3'))

          .map((n) => n.textContent.trim())

          .filter(Boolean)

          .slice(0, 25)

          .join('\n');

        return {

          ok: true,

          title,

          url: res.url || url,

          content: paragraphs || text.slice(0, 3000),

        };

      } catch (error) {

        return { ok: false, error: `${error.name}: ${error.message}` };

      }

    }

    async function searchHtml(query) {

      try {

        if (!query.trim()) return { ok: false, error: 'Truy vấn rỗng.' };

        const endpoint = `https://html.duckduckgo.com/html/?q=${encodeURIComponent(query)}`;

        const res = await fetch(endpoint, { mode: 'cors' });

        const text = await res.text();

        const doc = new DOMParser().parseFromString(text, 'text/html');

        const results = Array.from(doc.querySelectorAll('.result')).slice(0, 5).map((item) => {

          const title = item.querySelector('.result__title')?.textContent?.trim() || '';

          const snippet = item.querySelector('.result__snippet')?.textContent?.trim() || '';

          const url = item.querySelector('.result__title a')?.href || '';

          return { title, snippet, url };

        });

        return { ok: true, results };

      } catch (error) {

        return { ok: false, error: `${error.name}: ${error.message}` };

      }

    }

    function buildWebContext(query, urls = []) {

      return (async () => {

        const parts = [];

        if (query && query.trim()) {

          const search = await searchHtml(query);

          if (search.ok && search.results?.length) {

            parts.push(

              'SEARCH_RESULTS:\n' + search.results.map((item) =>

                `- ${item.title} | ${item.snippet} | ${item.url}`

              ).join('\n')

            );

          } else {

            parts.push(`SEARCH_ERROR: ${search.error || 'no results'}`);

          }

        }

        for (const url of urls) {

          const cached = state.webCache[url];

          if (cached && (Date.now() - cached.ts) < 300000) {

            parts.push(`URL_FETCH:\nTITLE: ${cached.title}\nURL: ${cached.url}\nCONTENT:\n${cached.content}`);

            continue;

          }

          const fetched = await fetchUrl(url);

          if (fetched.ok) {

            state.webCache[url] = { ...fetched, ts: Date.now() };

            parts.push(`URL_FETCH:\nTITLE: ${fetched.title}\nURL: ${fetched.url}\nCONTENT:\n${fetched.content}`);

          } else {

            parts.push(`URL_ERROR(${url}): ${fetched.error}`);

          }

        }

        persistState();

        return parts.join('\n\n').trim();

      })();

    }

    function extractTextFromResponse(payload) {

      if (payload.reply) return String(payload.reply);

      return JSON.stringify(payload, null, 2);

    function normalizeAssistantOutput(raw) {

      try {

        if (!raw) return { reply: '', code: '', analysis: '' };

        if (typeof raw === 'object' && raw.reply) {

          return {

            reply: String(raw.reply || ''),

            code: String(raw.code || ''),

            analysis: String(raw.analysis || ''),

          };

        }

        const text = String(raw);

        const fenced = text.match(/```(?:\w+)?\n([\s\S]*?)```/);

        const code = fenced ? fenced[1].trim() : '';

        return {

          reply: text.replace(/```[\s\S]*?```/g, '').trim(),

          code,

          analysis: '',

        };

      } catch (_) {

        return { reply: String(raw || ''), code: '', analysis: '' };

      }

    }

    async function hfFetch(modelName, messagePayload) {

      const res = await fetch(state.endpoint, {

        method: 'POST',

        headers: {

          'Content-Type': 'application/json',

          'Authorization': `Bearer ${state.token}`,

        },

        body: JSON.stringify({

          model: modelName,

          messages: messagePayload,

          max_tokens: 1800,

          temperature: Number($('tempInput').value || 0.35),

          stream: false,

        }),

      });

      const contentType = res.headers.get('content-type') || '';

      const bodyText = contentType.includes('application/json') ? JSON.stringify(await res.json()) : await res.text();

      if (!res.ok) {

        const err = new Error(`HTTP ${res.status}: ${bodyText.slice(0, 600)}`);

        err.status = res.status;

        err.body = bodyText;

        throw err;

      }

      return contentType.includes('application/json') ? JSON.parse(bodyText) : bodyText;

    }

    async function callModelWithFallback(messages, modelList) {

      let lastError = null;

      for (const model of modelList) {

        try {

          const result = await hfFetch(model, messages);

          const text = extractTextFromResponse(result);

          if (!text.trim()) {

            throw new Error('Empty response from model.');

          }

          setModelText(model);

          return { ok: true, model, raw: result, text };

        } catch (error) {

          lastError = {

            model,

            name: error.name || 'Error',

            message: error.message || String(error),

            status: error.status || null,

            body: error.body || '',

          };

          if (String(error.status) !== '404' && String(error.status) !== '503') {

            continue;

          }

        }

      }

      return {

        ok: false,

        error: lastError || { name: 'Error', message: 'Unknown error' },

      };

    }

    async function sendMessage(customText = null, isRetry = false) {

        persistState();

        pushMessage('user', text);

        const urls = parseUrls($('urlsInput').value);

        state.lastUrls = urls;

        const searchQuery = $('searchInput').value.trim() || text;

        state.lastSearch = searchQuery;

        const webContext = await buildWebContext(searchQuery, urls);

        const mode = $('modeSelect').value;

        state.lastMode = mode;

        state.lastPrompt = text;

        const userPrompt = buildPrompt(text, mode, webContext);

        const messages = [

          { role: 'system', content: 'You are a helpful, concise, production-grade assistant.' },

          ...buildMessagesForApi(),

          { role: 'user', content: userPrompt },

        ];

        const modelList = [state.modelName || MODEL_NAME, state.fallbackModel || FALLBACK_MODEL].filter(Boolean);

        const result = await callModelWithFallback(messages, modelList);

        if (!result.ok) {

          const analysis = errorAnalysis(result.error, modelList);

          const reply = `${analysis}\n\n### Hành động\n- Nhấn **Thử lại**.\n- Hoặc đổi token/model trong Settings.`;

          state.history.push({ role: 'assistant', content: reply });

          pushMessage('assistant', reply, { error: true, retry: text });

          renderRetryBlock(text);

          setDiagnostics(`<div class="error-box"><p class="error-title">API lỗi nhưng trang không trắng.</p><div>${renderMarkdownLite(analysis)}</div><button class="btn secondary" type="button" id="retryInline">Thử lại</button></div>`);

          const retryInline = $('retryInline');

          if (retryInline) retryInline.onclick = () => sendMessage(text, true);

          renderCodePreview('/* Model error. No code returned. */');

          persistState();

          return;

        }

        const normalized = normalizeAssistantOutput(result.text);

        const reply = normalized.reply || result.text;

        state.history.push({ role: 'assistant', content: reply });

        persistState();

        pushMessage('assistant', reply);

        if (normalized.code) {

          renderCodePreview(normalized.code);

        } else if (result.text.includes('```')) {

          const match = result.text.match(/```(?:\w+)?\n([\s\S]*?)```/);

          renderCodePreview(match ? match[1].trim() : result.text);

        } else {

          renderCodePreview(result.text);

        }

        const diag = [

          '## Runtime Diagnostics',

          `- Selected model: \`${result.model}\``,

          `- Mode: \`${mode}\``,

          `- Token present: \`${Boolean(state.token)}\``,

          `- Web URLs: \`${urls.length ? urls.join(', ') : 'none'}\``,

          `- Search: \`${searchQuery ? searchQuery.slice(0, 120) : 'none'}\``,

        ].join('\n');

        setDiagnostics(renderMarkdownLite(diag));

      } catch (error) {

        const analysis = errorAnalysis(error, [state.modelName, state.fallbackModel].filter(Boolean));

        const reply = `${analysis}\n\n### Hành động\n- Kiểm tra token / endpoint.\n- Nhấn **Thử lại** sau vài giây.`;

        state.history.push({ role: 'assistant', content: reply });

        persistState();

        pushMessage('assistant', reply, { error: true, retry: customText || $('promptInput').value });

        setDiagnostics(`<div class="error-box"><p class="error-title">Self-Healing Report</p><div>${renderMarkdownLite(analysis)}</div><button class="btn secondary" type="button" id="retryInline2">Thử lại</button></div>`);

        const retryInline = $('retryInline2');

        if (retryInline) retryInline.onclick = () => sendMessage(customText || $('promptInput').value, true);

        renderCodePreview(`/* ${error.name || 'Error'}: ${error.message || String(error)} */`);

      } finally {

        state.busy = false;

        $('sendBtn').disabled = false;

        setTokenState();

      }

    }

    function loadPromptChips() {

      const wrap = $('promptChips');

      wrap.innerHTML = '';

      promptChips.forEach((text) => {

        const btn = document.createElement('button');

        btn.className = 'chip';

        btn.type = 'button';

        btn.textContent = text;

        btn.onclick = () => {

          $('promptInput').value = text;

          sendMessage(text);

        };

        wrap.appendChild(btn);

      });

    }

    function renderHistory() {

      $('chatLog').innerHTML = '';

      if (!state.history.length) {

        pushMessage('assistant', 'Dashboard đã sẵn sàng. Nhập token, rồi gửi prompt để bắt đầu.');

        return;

      }

      for (const item of state.history) {

        const role = item.role === 'user' ? 'user' : 'assistant';

        pushMessage(role, item.content, { error: item.role === 'assistant' && /Self-Healing Report|Lỗi chính/i.test(item.content || '') });

      }

    }

    function openOverlay() {

      $('overlay').classList.remove('hidden');

      $('app').classList.add('hidden');

    }

    function closeOverlay() {

      $('overlay').classList.add('hidden');

      $('app').classList.remove('hidden');

    }

    function syncInputs() {

      $('tokenInput').value = state.token || '';

      $('brainToken').value = state.token || '';

      $('modelInput').value = state.modelName || MODEL_NAME;

      $('fallbackInput').value = state.fallbackModel || FALLBACK_MODEL;

      $('endpointInput').value = state.endpoint || HF_ENDPOINT;

      $('modelNameInput').value = state.modelName || MODEL_NAME;

      $('fallbackModelInput').value = state.fallbackModel || FALLBACK_MODEL;

      $('activeModelText').textContent = state.modelName || MODEL_NAME;

      setTokenState();

    }

    function saveTokenFromInputs() {

      state.token = $('tokenInput').value.trim() || $('brainToken').value.trim();

      state.modelName = $('modelInput').value.trim() || MODEL_NAME;

      state.fallbackModel = $('fallbackInput').value.trim() || FALLBACK_MODEL;

      state.endpoint = $('endpointInput').value.trim() || HF_ENDPOINT;

      persistState();

      syncInputs();

      closeOverlay();

      renderHistory();

      setDiagnostics('## Runtime Diagnostics\n- Token saved in localStorage.\n- Dashboard unlocked.');

    }

    function saveSettings() {

      state.modelName = $('modelNameInput').value.trim() || MODEL_NAME;

      state.fallbackModel = $('fallbackModelInput').value.trim() || FALLBACK_MODEL;

      $('modelInput').value = state.modelName;

      $('fallbackInput').value = state.fallbackModel;

      persistState();

      setModelText(state.modelName);

      setDiagnostics(

        [

          '## Settings Saved',

          `- MODEL_NAME: \`${state.modelName}\``,

          `- Fallback: \`${state.fallbackModel}\``,

          `- Endpoint: \`${state.endpoint}\``,

        ].join('\n')

      );

    }

    function initEvents() {

      document.querySelectorAll('.tab').forEach((btn) => {

        btn.addEventListener('click', () => setActiveTab(btn.dataset.tab));

      });

      $('saveTokenBtn').addEventListener('click', saveTokenFromInputs);

      $('brainSaveBtn').addEventListener('click', saveTokenFromInputs);

      $('brainForgetBtn').addEventListener('click', () => {

        state.token = '';

        persistState();

        syncInputs();

        openOverlay();

        setDiagnostics('## Runtime Diagnostics\n- Token cleared.');

      });

      $('clearTokenBtn').addEventListener('click', () => {

        state.token = '';

        persistState();

        syncInputs();

        openOverlay();

      });

      $('useDemoBtn').addEventListener('click', () => {

        state.token = '';

        persistState();

        syncInputs();

        closeOverlay();

        setDiagnostics('## Demo Mode\n- Token not required until you hit the API.');

      });

      $('sendBtn').addEventListener('click', () => sendMessage());

      $('retryBtn').addEventListener('click', () => sendMessage(state.lastPrompt, true));

      $('promptInput').addEventListener('keydown', (event) => {

        if (event.key === 'Enter' && (event.ctrlKey || event.metaKey)) {

          event.preventDefault();

          sendMessage();

        }

      });

      $('modelNameInput').addEventListener('change', saveSettings);

      $('fallbackModelInput').addEventListener('change', saveSettings);

      $('modelInput').addEventListener('change', saveSettings);

      $('fallbackInput').addEventListener('change', saveSettings);

      $('endpointInput').addEventListener('change', () => {

        state.endpoint = $('endpointInput').value.trim() || HF_ENDPOINT;

        persistState();

      });

    }

    function boot() {

      loadState();

      syncInputs();

      loadPromptChips();

      initEvents();

      renderHistory();

      setActiveTab('workspace');

      if (state.token) {

        closeOverlay();

      } else {

        openOverlay();

      }

      setDiagnostics('## Runtime Diagnostics\n- Ready.');

      setTokenState();

      try {

        if (window.lucide && typeof window.lucide.createIcons === 'function') {

          window.lucide.createIcons();

        }

      } catch (error) {

        console.warn('lucide init failed', error);

      }

    }

    boot();
